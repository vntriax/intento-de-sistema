require_once 'config/database.php';
require_once 'config/constants.php';

$database = new Database();
$conn = $database->getConnection();
$calculado = false;
$resultado = null;

// Obtener empleados
$stmt = $conn->query("SELECT id, cedula, nombre, apellido FROM empleados WHERE estatus = 'ACTIVO' ORDER BY apellido, nombre");
$empleados = $stmt->fetchAll();

// Procesar cálculo
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['empleado_id'])) {
    $empleado_id = $_POST['empleado_id'];
    $anio_inicio = $_POST['anio_inicio'];
    $anio_fin = $_POST['anio_fin'];
    
    // Obtener datos del empleado
    $stmt = $conn->prepare("SELECT * FROM empleados WHERE id = ?");
    $stmt->execute([$empleado_id]);
    $empleado = $stmt->fetch();
    
    if ($empleado) {
        // Obtener historial salarial
        $stmt = $conn->prepare("SELECT * FROM historial_salarial WHERE empleado_id = ? ORDER BY fecha_inicio");
        $stmt->execute([$empleado_id]);
        $historial = $stmt->fetchAll();
        
        // Calcular alícuotas mensuales
        $alicuotas = [];
        $total_garantia = 0;
        $total_intereses = 0;
        
        for ($anio = $anio_inicio; $anio <= $anio_fin; $anio++) {
            for ($mes = 1; $mes <= 12; $mes++) {
                // Obtener salario del mes
                $salario_base = $empleado['salario_mensual'];
                
                // Calcular alícuota vacacional (15 días / 12 meses)
                $alicuota_vacacional = ($salario_base / 30) * 15 / 12;
                
                // Calcular alícuota aguinaldo (30 días / 12 meses)
                $alicuota_aguinaldo = ($salario_base / 30) * 30 / 12;
                
                // Salario integral
                $salario_integral = $salario_base + $alicuota_vacacional + $alicuota_aguinaldo + $empleado['prima_por_hijos'];
                
                // Garantía trimestral (15 días por trimestre)
                $garantia_trimestral = ($salario_integral / 30) * 15;
                
                // Obtener tasa del mes
                $stmt = $conn->prepare("SELECT tasa_promedio FROM tasas_interes_historicas WHERE anio = ? AND mes = ?");
                $stmt->execute([$anio, $mes]);
                $tasa_data = $stmt->fetch();
                $tasa_promedio = $tasa_data['tasa_promedio'] ?? TASA_PROMEDIO_ACTUAL;
                
                // Calcular intereses
                $intereses_mes = $garantia_trimestral * ($tasa_promedio / 100) / 12;
                
                $alicuotas[] = [
                    'anio' => $anio,
                    'mes' => $mes,
                    'salario_base' => $salario_base,
                    'alicuota_vacacional' => $alicuota_vacacional,
                    'alicuota_aguinaldo' => $alicuota_aguinaldo,
                    'prima_hijos' => $empleado['prima_por_hijos'],
                    'salario_integral' => $salario_integral,
                    'garantia' => $garantia_trimestral,
                    'tasa' => $tasa_promedio,
                    'intereses' => $intereses_mes
                ];
                
                $total_garantia += $garantia_trimestral;
                $total_intereses += $intereses_mes;
            }
        }
        
        $resultado = [
            'empleado' => $empleado,
            'periodo' => [
                'inicio' => $anio_inicio,
                'fin' => $anio_fin
            ],
            'alicuotas' => $alicuotas,
            'totales' => [
                'garantia' => $total_garantia,
                'intereses' => $total_intereses,
                'total' => $total_garantia + $total_intereses
            ]
        ];
        
        $calculado = true;
        
        // Guardar en base de datos
        $stmt = $conn->prepare("INSERT INTO calculos_fideicomiso 
            (empleado_id, fecha_calculo, periodo_inicio, periodo_fin, salario_base,
            total_garantia_prestaciones, total_intereses_garantia, total_general)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)");
        $stmt->execute([
            $empleado_id, date('Y-m-d'), "$anio_inicio-01-01", "$anio_fin-12-31",
            $empleado['salario_mensual'], $total_garantia, $total_intereses, $total_garantia + $total_intereses
        ]);
    }
}
