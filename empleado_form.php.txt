        'estatus' => $_POST['estatus']
    ];

        // Actualizar
        $stmt = $conn->prepare("UPDATE empleados SET 
            cedula = ?, nombre = ?, apellido = ?, cargo = ?, tipo_personal = ?,
            fecha_ingreso = ?, fecha_egreso = ?, motivo_egreso = ?, salario_mensual = ?,
            prima_antiguedad = ?, prima_profesional = ?, prima_por_hijos = ?,
            hijos_cantidad = ?, fecha_inicio_prima_hijos = ?, estatus = ?

            $_POST['id']
        ]);

        // Insertar
        $stmt = $conn->prepare("INSERT INTO empleados 
            (cedula, nombre, apellido, cargo, tipo_personal, fecha_ingreso, fecha_egreso, motivo_egreso,
            salario_mensual, prima_antiguedad, prima_profesional, prima_por_hijos, hijos_cantidad,
            fecha_inicio_prima_hijos, estatus) 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
        $stmt->execute([
            $datos['cedula'], $datos['nombre'], $datos['apellido'], $datos['cargo'], $datos['tipo_personal'],
            $datos['fecha_ingreso'], $datos['fecha_egreso'], $datos['motivo_egreso'], $datos['salario_mensual'],
            $datos['prima_antiguedad'], $datos['prima_profesional'], $datos['prima_por_hijos'],
            $datos['hijos_cantidad'], $datos['fecha_inicio_prima_hijos'], $datos['estatus']
        ]);
    }
    
    $mensaje = '<div class="alert alert-success"><i class="fas fa-check-circle"></i> Empleado guardado exitosamente.</div>';
    
    if (!isset($_POST['id']) || empty($_POST['id'])) {
        $empleado_id = $conn->lastInsertId();
        header("Location: empleado_form.php?id=$empleado_id");
        exit;
    }
