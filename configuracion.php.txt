require_once 'config/database.php';
require_once 'config/constants.php';

$database = new Database();
$conn = $database->getConnection();
$mensaje = '';

// Procesar formulario
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $nombre_ente = $_POST['nombre_ente'];
    $rif = $_POST['rif'];
    $direccion = $_POST['direccion'];
    $telefono = $_POST['telefono'];
    $email = $_POST['email'];
    $nombre_alcaldia = $_POST['nombre_alcaldia'];
    $nombre_fundacion = $_POST['nombre_fundacion'];
    $departamento = $_POST['departamento'];
    
    // Manejo de logo
    $logo_path = $_POST['logo_path_actual'];
    if (isset($_FILES['logo']) && $_FILES['logo']['error'] === 0) {
        $upload_dir = 'uploads/logos/';
        if (!file_exists($upload_dir)) {
            mkdir($upload_dir, 0777, true);
        }
        $logo_path = $upload_dir . time() . '_' . $_FILES['logo']['name'];
        move_uploaded_file($_FILES['logo']['tmp_name'], $logo_path);
    }
    
    // Verificar si existe configuración
    $stmt = $conn->query("SELECT COUNT(*) as count FROM configuracion_ente");
    $count = $stmt->fetch()['count'];
    
    if ($count > 0) {
        $stmt = $conn->prepare("UPDATE configuracion_ente SET 
            nombre_ente = ?, rif = ?, direccion = ?, telefono = ?, email = ?, 
            logo_path = ?, nombre_alcaldia = ?, nombre_fundacion = ?, departamento = ? 
            WHERE id = 1");
        $stmt->execute([$nombre_ente, $rif, $direccion, $telefono, $email, $logo_path, $nombre_alcaldia, $nombre_fundacion, $departamento]);
    } else {
        $stmt = $conn->prepare("INSERT INTO configuracion_ente 
            (nombre_ente, rif, direccion, telefono, email, logo_path, nombre_alcaldia, nombre_fundacion, departamento) 
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)");
        $stmt->execute([$nombre_ente, $rif, $direccion, $telefono, $email, $logo_path, $nombre_alcaldia, $nombre_fundacion, $departamento]);
    }
    
    $mensaje = '<div class="alert alert-success"><i class="fas fa-check-circle"></i> Configuración guardada exitosamente.</div>';
}
