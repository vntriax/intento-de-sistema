<?php
// prestaciones_calculo.php
require_once 'config/database.php';
require_once 'config/constants.php';

$database = new Database();
$conn = $database->getConnection();
$calculado = false;
$resultado = null;

// Obtener empleados para el selector
$stmt = $conn->query("SELECT id, cedula, nombre, apellido FROM empleados WHERE estatus = 'ACTIVO' ORDER BY apellido, nombre");
$empleados = $stmt->fetchAll();

// Procesar cálculo
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['empleado_id'])) {
    $empleado_id = $_POST['empleado_id'];
    $fecha_egreso = $_POST['fecha_egreso'];
    
    // Obtener datos del empleado
    $stmt = $conn->prepare("SELECT * FROM empleados WHERE id = ?");
    $stmt->execute([$empleado_id]);
    $empleado = $stmt->fetch();
    
    if ($empleado) {
        // Calcular tiempo de servicio
        $fecha_ingreso = new DateTime($empleado['fecha_ingreso']);
        $fecha_fin = new DateTime($fecha_egreso);
        $diferencia = $fecha_ingreso->diff($fecha_fin);
        
        $anios = $diferencia->y;
        $meses = $diferencia->m;
        $dias = $diferencia->d;
        
        // Calcular salario diario e integral
        $salario_mensual = $empleado['salario_mensual'] + $empleado['prima_antiguedad'] + $empleado['prima_profesional'] + $empleado['prima_por_hijos'];
        $salario_diario = $salario_mensual / 30;
        $salario_integral = $salario_diario + ($empleado['prima_por_hijos'] > 0 ? 0 : 0); // Ajustar según normativa
        
        // Cálculo Prestaciones Sociales - Artículo 142 Literal A y B [[51]]
        // Garantía de prestaciones: 15 días por trimestre
        $total_trimestres = floor(($anios * 12 + $meses) / 3);
        $garantia_prestaciones = $salario_diario * 15 * $total_trimestres;
        
        // Intereses sobre garantía (Artículo 143 LOTTT) [[5]]
        // Obtener tasa promedio del último periodo
        $stmt = $conn->query("SELECT tasa_promedio FROM tasas_interes_historicas ORDER BY anio DESC, mes DESC LIMIT 1");
        $tasa = $stmt->fetch();
        $tasa_promedio = $tasa['tasa_promedio'] ?? TASA_PROMEDIO_ACTUAL;
        
        $intereses_garantia = $garantia_prestaciones * ($tasa_promedio / 100) * ($anios + $meses/12);
        
        // Total prestaciones sociales
        $total_prestaciones = $garantia_prestaciones + $intereses_garantia;
        
        // Otros beneficios
        // Bono vacacional fraccionado
        $dias_vacaciones_fraccion = ($meses * 15) / 12;
        $bono_vacacional_fraccion = $salario_diario * $dias_vacaciones_fraccion;
        
        // Bono fin de año fraccionado
        $dias_bono_anio_fraccion = ($meses * 30) / 12;
        $bono_fin_anio_fraccion = $salario_diario * $dias_bono_anio_fraccion;
        
        // Vacaciones no disfrutadas
        $vacaciones_no_disfrutadas = $salario_diario * 15 * $anios;
        
        // Fracción de sueldos y salarios
        $fraccion_sueldos = $salario_diario * $dias;
        
        // Fracción bono alimentación
        $bono_alimentacion_diario = 208.50; // Ajustar según gaceta
        $fraccion_bono_alimentacion = $bono_alimentacion_diario * $dias;
        
        // Subtotal otros beneficios
        $subtotal_otros_beneficios = $bono_vacacional_fraccion + $bono_fin_anio_fraccion + 
                                     $vacaciones_no_disfrutadas + $fraccion_sueldos + $fraccion_bono_alimentacion;
        
        // Total asignaciones
        $total_asignaciones = $total_prestaciones + $subtotal_otros_beneficios;
        
        // Cálculo Artículo 142 Literal C [[50]]
        $prestaciones_30_dias = $salario_diario * 30 * $anios;
        if ($meses > 6) {
            $prestaciones_30_dias += $salario_diario * 30;
        }
        
        $resultado = [
            'empleado' => $empleado,
            'fecha_ingreso' => $empleado['fecha_ingreso'],
            'fecha_egreso' => $fecha_egreso,
            'tiempo_servicio' => [
                'anios' => $anios,
                'meses' => $meses,
                'dias' => $dias
            ],
            'salarios' => [
                'mensual' => $salario_mensual,
                'diario' => $salario_diario,
                'integral' => $salario_integral
            ],
            'prestaciones_sociales' => [
                'garantia' => $garantia_prestaciones,
                'intereses' => $intereses_garantia,
                'total' => $total_prestaciones
            ],
            'otros_beneficios' => [
                'bono_vacacional' => $bono_vacacional_fraccion,
                'bono_fin_anio' => $bono_fin_anio_fraccion,
                'vacaciones_no_disfrutadas' => $vacaciones_no_disfrutadas,
                'fraccion_sueldos' => $fraccion_sueldos,
                'fraccion_bono_alimentacion' => $fraccion_bono_alimentacion,
                'subtotal' => $subtotal_otros_beneficios
            ],
            'totales' => [
                'total_asignaciones' => $total_asignaciones,
                'deducciones' => 0,
                'total_pagar' => $total_asignaciones
            ],
            'articulo_142_c' => [
                'prestaciones_30_dias' => $prestaciones_30_dias,
                'garantia_depositada' => $garantia_prestaciones,
                'total_general' => $prestaciones_30_dias
            ],
            'tasa_aplicada' => $tasa_promedio
        ];
        
        $calculado = true;
        
        // Guardar en base de datos
        $stmt = $conn->prepare("INSERT INTO calculos_prestaciones 
            (empleado_id, fecha_calculo, fecha_ingreso, fecha_egreso, tiempo_servicio_anios, 
            tiempo_servicio_meses, tiempo_servicio_dias, salario_mensual, salario_diario,
            prestaciones_sociales_art142_ab, intereses_garantia_prestaciones,
            fraccion_bono_vacacional, fraccion_bono_fin_anio, vacaciones_no_disfrutadas,
            fraccion_sueldos, fraccion_bono_alimentacion, total_asignaciones, total_pagar,
            prestaciones_30_dias, garantia_depositada, total_general_142c, tasa_promedio_aplicada)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
        $stmt->execute([
            $empleado_id, date('Y-m-d'), $empleado['fecha_ingreso'], $fecha_egreso,
            $anios, $meses, $dias, $salario_mensual, $salario_diario,
            $garantia_prestaciones, $intereses_garantia,
            $bono_vacacional_fraccion, $bono_fin_anio_fraccion, $vacaciones_no_disfrutadas,
            $fraccion_sueldos, $fraccion_bono_alimentacion, $total_asignaciones, $total_asignaciones,
            $prestaciones_30_dias, $garantia_prestaciones, $prestaciones_30_dias, $tasa_promedio
        ]);
    }
}
